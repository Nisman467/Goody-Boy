class Generator
{

	char digit[11];
	char symbol[9];
	
	public:
		void password_Generator(char *password, char useUpCase, char useLowCase, char useDigit, char useSymbol, int length)
		{
			char clonePass[1000]="\0";
			int randValue;
			
			if(useUpCase =='Y' || useUpCase =='y')
			{
				strcpy(uppercase,"ABCDEFGHIJKLMNOPQRSTUVWXYZ");
				strcat(clonePass,uppercase);			//concatenation of a required words
			}
			if(useLowCase =='Y' || useLowCase =='y')
			{
				strcpy(lowercase,"abcdefghijklmnopqrstuvwxyz");
				strcat(clonePass,lowercase);			//concatenation of a required words
			}
			if(useDigit =='Y' || useDigit =='y')
			{
				strcpy(digit,"0123456789");
				strcat(clonePass,digit);			//concatenation of a required words
			}
			if(useSymbol =='Y' || useSymbol =='y')
			{
				strcpy(symbol,"!@#$%^&_");	
				strcat(clonePass,symbol);			//concatenation of a required words
			}
				
		
			
			//generation of password
			for(int i=0; i<length; i++)
			{
				randValue =rand() % strlen(clonePass);
				password[i]= clonePass[randValue];
			}
			
			password[length]='\0';
		}
};
